ClipFactory — Pipeline

one long video → many short, vertical, captioned clips · ffmpeg + Whisper + LLM + face-aware framing

flowchart TD
  sched["⏰ daily 09:30 (scheduled)\nor: node pipeline.mjs"]:::sch --> A

  subgraph S1["1 · SOURCE  (source.mjs)"]
    direction TB
    A["sources.json\nYouTube channels (@handle / UC-id)"]:::io --> B["resolveChannelId\n@handle → channel UC-id"]
    B --> C["channelFeed\nfree RSS: youtube.com/feeds/videos.xml\n(no API key, not bot-blocked)"]
    C --> D["filter unseen (.source-seen.json) + recent\nscoreTitle by hook words / numbers / '?'"]
    D --> E[("queue.json\nranked candidates")]:::io
  end

  E --> LOOP{"for each candidate\nuntil N clipped"}

  subgraph S2["2 · DOWNLOAD  (pipeline.mjs)"]
    direction TB
    LOOP --> G["yt-dlp · player_client=android_vr,web_safari\n720p DASH (video+audio merge)\nbypasses YouTube n-challenge"]
    G -. "403 / blocked" .-> LOOP
    G --> H["trimHead → first 8 min\n(bounds Whisper memory)"]
  end

  subgraph S3["3 · CLIP ENGINE  (clip.mjs · makeClips)"]
    direction TB
    H --> I["transcribe · Whisper (base, word timestamps)\n→ cues (segments) + words (per-word times)"]
    I --> J["buildWindows · sentence-aware\nstart @ thought-start · end @ sentence-end · 14–44s"]
    J --> K["scoreWindow (heuristic) → prefilter top 10"]
    K --> L["rankWindowsLLM  ★ DECISION ENGINE\nscore virality 0–100 + write a hook"]
    L --> L1{"local Gemma\n(Ollama, CPU, free)"}
    L1 -- "ok" --> M
    L1 -- "OOM / down" --> L2["☁ Gemini cloud\n(gemini-flash-latest)"]:::cloud
    L2 --> M["rank by viral score → pick top N"]
  end

  subgraph S4["4 · RENDER  (renderClip · per clip)"]
    direction TB
    M --> N["buildAss\nword-by-word karaoke captions (yellow fill)\n+ hook title card (first 3s)"]
    M --> O["faceInfo · OpenCV (face_track.py)\nsample frames → frac, face x"]
    O --> P{"face in ≥15%\nof frames?"}
    P -- "yes" --> Q["crop around speaker\n(x = face centre)"]
    P -- "no (B-roll)" --> R["blur-fit\nwhole frame + blurred bg"]
    N --> FF
    Q --> FF["ffmpeg · cut [start,end]\n→ 1080×1920 → burn captions\nlibx264 / aac"]
    R --> FF
    FF --> T["clip-N.mp4  +  clips.json\n(viral score · hook · dur)"]:::io
  end

  T --> Z["markSeen → deliver clips\n(Telegram / folder)"]:::done
